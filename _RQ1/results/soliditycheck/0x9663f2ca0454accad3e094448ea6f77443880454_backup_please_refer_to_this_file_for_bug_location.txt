903	 function binarySearch( Observation[65535] storage self, uint32 time, uint32 target, uint16 index, uint16 cardinality ) private view returns (Observation memory beforeOrAt, Observation memory atOrAfter) {
904	 uint256 l = (index + 1) % cardinality;
905	 uint256 r = l + cardinality - 1;
906	 uint256 i;
907	 while (true) {
908	 i = (l + r) / 2;
909	 beforeOrAt = self[i % cardinality];
910	 if (!beforeOrAt.initialized) {
911	 l = i + 1;
912	 continue;
913	 }
914	 atOrAfter = self[(i + 1) % cardinality];
915	 bool targetAtOrAfter = lte(time, beforeOrAt.blockTimestamp, target);
916	 if (targetAtOrAfter && lte(time, target, atOrAfter.blockTimestamp)) break;
917	 if (!targetAtOrAfter) r = i - 1;
918	 else l = i + 1;
919	 }

986	 function getNextSqrtPriceFromAmount0RoundingUp( uint160 sqrtPX96, uint128 liquidity, uint256 amount, bool add ) internal pure returns (uint160) {
987	 if (amount == 0) return sqrtPX96;
988	 uint256 numerator1 = uint256(liquidity) << FixedPoint96.RESOLUTION;
989	 if (add) {
990	 uint256 product;
991	 if ((product = amount * sqrtPX96) / amount == sqrtPX96) {
992	 uint256 denominator = numerator1 + product;
993	 if (denominator >= numerator1) return uint160(FullMath.mulDivRoundingUp(numerator1, sqrtPX96, denominator));
994	 }
995	 return uint160(UnsafeMath.divRoundingUp(numerator1, (numerator1 / sqrtPX96).add(amount)));
996	 }
997	 else {
998	 uint256 product;
999	 require((product = amount * sqrtPX96) / amount == sqrtPX96 && numerator1 > product);
1000	 uint256 denominator = numerator1 - product;
1001	 return FullMath.mulDivRoundingUp(numerator1, sqrtPX96, denominator).toUint160();
1002	 }
1003	 }
1004	 function getNextSqrtPriceFromAmount1RoundingDown( uint160 sqrtPX96, uint128 liquidity, uint256 amount, bool add ) internal pure returns (uint160) {
1005	 if (add) {
1006	 uint256 quotient = ( amount <= type(uint160).max ? (amount << FixedPoint96.RESOLUTION) / liquidity : FullMath.mulDiv(amount, FixedPoint96.Q96, liquidity) );
1007	 return uint256(sqrtPX96).add(quotient).toUint160();
1008	 }
1009	 else {
1010	 uint256 quotient = ( amount <= type(uint160).max ? UnsafeMath.divRoundingUp(amount << FixedPoint96.RESOLUTION, liquidity) : FullMath.mulDivRoundingUp(amount, FixedPoint96.Q96, liquidity) );
1011	 require(sqrtPX96 > quotient);
1012	 return uint160(sqrtPX96 - quotient);
1013	 }
1014	 }
